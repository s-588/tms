package http

func (s *Server) setHandlers() {
	s.setClientEndpoints()
	s.setEmployeeEndpoints()
	s.setOrderEndpoints()
	s.setPriceEndpoints()
	s.setTransportEndpoints()
	s.setInspectionEndpoints()
	s.setInsuranceEndpoints()
	s.setNodeEndpoints()
	s.setSearchEndpoint()
}

func (s *Server) setClientEndpoints() {
	s.mux.HandleFunc("GET /clients/page", s.Handler.GetClientsPage)
	s.mux.HandleFunc("GET /clients", s.Handler.GetClients)
	s.mux.HandleFunc("GET /clients/{id}", s.Handler.GetClientHandler)
	// s.mux.HandleFunc("GET /clients/new", s.Handler.NewClientPageHandler)        // if exists
	// s.mux.HandleFunc("GET /clients/{id}/edit", s.Handler.EditClientPageHandler) // if exists
	s.mux.HandleFunc("POST /clients", s.Handler.CreateClientHandler)
	s.mux.HandleFunc("PUT /clients/{id}", s.Handler.UpdateClient)
	s.mux.HandleFunc("DELETE /clients/{id}", s.Handler.DeleteClient)
	s.mux.HandleFunc("POST /clients/delete", s.Handler.BulkDeleteClients)
	s.mux.HandleFunc("GET /clients/{id}/orders", s.Handler.GetClientOrders)
}
